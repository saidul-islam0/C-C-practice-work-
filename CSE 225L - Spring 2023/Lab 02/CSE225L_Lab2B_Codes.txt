#include <iostream>
using namespace std;

int main() {
    // ask size from user
    int m, n;
    cout << "Enter row size, m: ";
    cin >> m;
    cout << "Enter column size, n: ";
    cin >> n;
    
    // dynamically allocate 2D list 
    int **arr = NULL;
    arr = new int*[m];  // allocate rows
    // int **arr = new int*[m];  // also ok
    
    // allocate cols for each row
    for(int i=0; i<m; i++) {
        arr[i] = new int[n];    
    }
    
    // take user input for list 
    cout << "Enter " << (m*n) << " elements: " << endl;
    for(int i=0; i<m; i++) {
        for(int j=0; j<n; j++) {
            cin >> arr[i][j];
        }
    }
    
    // print the list 
    cout << "Print " << (m*n) << " elements: " << endl;
    for(int i=0; i<m; i++) {
        for(int j=0; j<n; j++) {
            cout << arr[i][j] << " ";
        }
        cout << endl;
    }
    
    // deallocate memory spaces
    for(int i=0; i<m; i++) {
        delete[] arr[i];
    }
    delete[] arr;

    return 0;
}
